## 一般的插件封装规范结构

```js
(function(global) {
  'use strict';
  function MyPlugin(el, options) {
    //some cod
  }
  MyPlugin.prototype = {
    //定义方法
    show: function() {
      //some code
    }
  };
  if (typeof module !== 'undefined' && module.exports) {
    //兼容 CommonJs 规范
    module.exports = MyPlugin;
  } else if (typeof define === 'function') {
    //兼容 AMD/CMD 规范
    define(function() {
      return MyPlugin;
    });
  } else {
    //注册全局变量，兼容直接使用 script 标签引入插件
    global.MyPlugin = MyPlugin;
  }
})(this);
```
