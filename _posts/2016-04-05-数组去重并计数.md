 var arr = [1,23,34,2,2,323,4,34,56,1,23,12,23,23,32,332,34,];

        Array.prototype.unique = function (){
            var length = this.length;//获取数组内容长度
            var obj = {
                newArr : []
            };
       
            for (var i = 0; i < length; i++ ) {
                if (!obj[this[i]]) {
                    obj.newArr.push(this[i]);
                    obj[this[i]] = 1;
                } else {
                    obj[this[i]] ++;
                }
            }
            return obj ;
        }
